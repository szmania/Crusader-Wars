control_toggle_1 = {
	scope = character

	is_shown = {
		has_variable = control_toggle_1
	}

	effect = {
		if = {
			limit = {
				has_variable = control_toggle_1
			}
			remove_variable = control_toggle_1
		}
		else = {
			set_variable = control_toggle_1
		}
	}
}
